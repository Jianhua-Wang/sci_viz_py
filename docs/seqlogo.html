import seqlogo
def logo(a):
    pfm = a.copy()
    a = pfm.sum(axis=1)
    for allele in 'ACGT':
        pfm[allele] = pfm[allele]/a
    pwm = seqlogo.pfm2pwm(pfm.values)
    pwm = seqlogo.Pwm(pwm)
    ppm = seqlogo.pwm2ppm(pwm)
    ppm = seqlogo.Ppm(ppm.values)
    a = seqlogo.seqlogo(ppm, ic_scale = True, format = 'png', size = 'large')
    cpm = seqlogo.CompletePm(ppm = ppm)
    print(cpm.consensus)
    return a

logo(pfm)
